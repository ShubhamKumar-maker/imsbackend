create table jobOpenings (
requirementId bigint primary key AUTO_INCREMENT,
requirementDetails varchar(20) not null,
startDate date not null,
enddate date not null,
skillid int not null,
noOfCandidates int not null);


create table panelavilability(
panelavilabilityID bigint primary key auto_increment,
